        let currentCards = [], curIdx = 0;

        function loadChapter() {
            const chap = document.getElementById('chapMenu').value;
            db.ref('finance_master/' + chap).on('value', (snap) => {
                currentCards = [];
                snap.forEach(child => {
                    currentCards.push({id: child.key, ...child.val()});
                });
                
                if(currentCards.length === 0 && MASTER_DATA[chap]) {
                    MASTER_DATA[chap].forEach(item => {
                        db.ref('finance_master/' + chap).push(item);
                    });
                }
                curIdx = 0;
                updateUI();
            });
        }

        function updateUI() {
            if(currentCards.length > 0) {
                document.getElementById('qText').innerText = currentCards[curIdx].q;
                document.getElementById('aText').innerText = currentCards[curIdx].a;
                document.getElementById('countDisplay').innerText = `${curIdx + 1} / ${currentCards.length}`;
            }
            document.getElementById('mainCard').classList.remove('is-flipped');
        }
